$(document).ready(function() {
    $('#example').dataTable({
        "columnDefs": [{
            "render": function(data, type, row) {
                return data + ' (' + row[3] + ')';
            },
            "targets": 0
        },
        {
            "visible": false,
            "targets": [3]
        }]
    });
});
